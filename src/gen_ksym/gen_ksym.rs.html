1//! A binary to generate compressed kallsyms blob from symbol list input (format: `address type name` per line, e.g. `ffffffff81000000 T _stext`).
2//! The output blob can be directly used in the kernel or for offline analysis.
3//!

10/// Candidate prefix lengths for heuristic tokenization
11// const PREFIX_CANDIDATE_LENS: &[usize] = &[8, 16, 24, 32, 64, 96, 128, 512, 1024, 2048];
12const PREFIX_CANDIDATE_LENS: &[usize] = &[
13    10, 24, 31, 40, 56, 60, 70, 80, 90, 100, 150, 200, 250, 300, 400, 500, 600, 700, 800, 900,
14    1000, 1200, 1400, 1600, 1800, 2000,
15];
16
17/// Maximum number of tokens
18const MAX_TOKEN: usize = 512;
19
20/// The structure for compressed symbol data
21pub struct KallsymsBlob {
22    /// The token table storing all tokens as concatenated bytes
23    pub token_table: Vec<u8>,
24    /// The start index of each token in token_table
25    pub token_index: Vec<u32>,
26    /// A map from token string to its ID (index in token_index)
27    pub token_map: HashMap<String, u16>,
28    /// Compressed symbol data
29    pub kallsyms_names: Vec<u8>,
30    /// The offsets of each symbol in kallsyms_names
31    pub kallsyms_offsets: Vec<u32>,
32    /// The sequence numbers of each symbol
33    pub kallsyms_seqs_of_names: Vec<u32>,
34    /// The addresses of each symbol
35    pub kallsyms_addresses: Vec<u64>,
36    /// The total number of symbols
37    pub kallsyms_num_syms: usize,
38}

66impl KallsymsBlob {
67    /// Create a new empty blob
68    pub fn new() -> Self {
69        Self {
70            token_table: Vec::new(),
71            token_index: Vec::new(),
72            token_map: HashMap::new(),
73            kallsyms_names: Vec::new(),
74            kallsyms_offsets: Vec::new(),
75            kallsyms_seqs_of_names: Vec::new(),
76            kallsyms_addresses: Vec::new(),
77            kallsyms_num_syms: 0,
78        }
79    }

81    /// Add a token to the token table
82    fn add_token(&mut self, token: String) -> Option<u16> {
83        if let Some(&id) = self.token_map.get(&token) {
84            return Some(id);
85        }
86        let id = self.token_map.len() as u16;
87        self.token_index.push(self.token_table.len() as u32);
88        self.token_table.extend_from_slice(token.as_bytes());
89        self.token_map.insert(token, id);
90        Some(id)
91    }

93    /// Compress all symbol names, auto-generating tokens.
94    /// Storage order: address order, with a name-order to address-order mapping.
95    pub fn compress_symbols(&mut self, symbols: &[(String, u64, char)]) {
96        // 0) Build indices for name order and address order.
97        let n = symbols.len();
98        if n == 0 {
99            return;
100        }
101        let mut idx_by_addr: Vec<usize> = (0..n).collect();
102        idx_by_addr.sort_by_key(|&i| symbols[i].1);
103        let mut idx_by_name: Vec<usize> = (0..n).collect();
104        idx_by_name.sort_by(|&i, &j| symbols[i].0.cmp(&symbols[j].0));
105
106        // map original index -> address-order index
107        let mut orig_to_addr_idx = vec![0usize; n];
108        for (addr_pos, &orig_idx) in idx_by_addr.iter().enumerate() {
109            orig_to_addr_idx[orig_idx] = addr_pos;
110        }
111
112        // 1) Count possible tokens using prefix-based heuristic.
113        // For each symbol, consider only prefixes of specific lengths.
114        let mut token_count: HashMap<String, usize> = HashMap::new();
115        for (name, _, _) in symbols.iter() {
116            let bytes = name.as_bytes();
117            for &len in PREFIX_CANDIDATE_LENS {
118                if bytes.len() >= len {
119                    let token = std::str::from_utf8(&bytes[..len]).unwrap();
120                    *token_count.entry(token.to_string()).or_insert(0) += 1;
121                } else if !bytes.is_empty() && len == PREFIX_CANDIDATE_LENS[0] {
122                    // Edge case: name shorter than the smallest candidate length; include full name to avoid missing short common prefixes
123                    let token = name;
124                    *token_count.entry(token.to_string()).or_insert(0) += 1;
125                }
126            }
127        }
128
129        // 2) Select high-frequency tokens (cap at MAX_TOKEN), prefer longer tokens on tie
130        let mut tokens: Vec<(String, usize)> = token_count.into_iter().collect();
131        tokens.sort_by(|a, b| {
132            // primary: frequency desc; secondary: length desc;
133            // b.1.cmp(&a.1).then_with(|| b.0.len().cmp(&a.0.len()))
134            (b.1 * b.0.len()).cmp(&(a.1 * a.0.len())) // weighted by length(more effective)
135        });
136        let mut final_token_list: Vec<String> = Vec::new();
137        for (tok, _) in tokens.into_iter() {
138            if final_token_list.len() >= MAX_TOKEN {
139                break;
140            }
141            // Avoid tokens that are prefixes of existing tokens
142            let mut is_prefix = false;
143            for existing in &final_token_list {
144                if existing.starts_with(&tok) {
145                    is_prefix = true;
146                    break;
147                }
148            }
149            if !is_prefix {
150                final_token_list.push(tok);
151            }
152        }
153        for tok in final_token_list.into_iter() {
154            self.add_token(tok);
155        }
156
157        // 3) Compress symbols in address order and build offsets/addresses.
158        for &orig_idx in &idx_by_addr {
159            let (ref sym, addr, ty) = symbols[orig_idx];
160            self.kallsyms_offsets.push(self.kallsyms_names.len() as u32);
161            self.kallsyms_addresses.push(addr);
162
163            let sym_bytes = sym.as_bytes();
164            // Only allow a single token at the beginning (prefix) according to heuristic.
165            let rem = sym_bytes.len();
166            let mut consumed = 0usize;
167            for &l in PREFIX_CANDIDATE_LENS.iter().rev() {
168                if l > rem {
169                    // if the candidate length exceeds the remaining length, skip
170                    continue;
171                }
172                let candidate = &sym[..l];
173                if let Some(&id) = self.token_map.get(candidate) {
174                    // 1. [type] [length] (0xff  token             0xff) [remaining bytes]
175                    // 2. [type] [length] (0xff  token_hi token_lo 0xff) [remaining bytes]
176                    //    [1byte][2bytes] (1byte 1byte    1byte   1byte) [remaining bytes]
177                    let mut length: u16 = if id < 256 { 3 } else { 4 };
178                    length += (rem - l) as u16;
179                    // Emit type char
180                    self.kallsyms_names.push(ty as u8);
181
182                    // Emit length (little-endian: lo, hi)
183                    self.kallsyms_names.push((length & 0xFF) as u8);
184                    self.kallsyms_names.push((length >> 8) as u8);
185
186                    // Emit token
187                    self.kallsyms_names.push(TOKEN_MARKER);
188                    if id < 256 {
189                        self.kallsyms_names.push(id as u8);
190                    } else {
191                        self.kallsyms_names.push((id >> 8) as u8);
192                        self.kallsyms_names.push((id & 0xFF) as u8);
193                    }
194                    self.kallsyms_names.push(TOKEN_MARKER);
195                    consumed = l;
196                    break;
197                }
198            }
199
200            if consumed == 0 {
201                // No token matched; emit full symbol as raw bytes
202                // Emit type char
203                self.kallsyms_names.push(ty as u8);
204                // Emit length
205                let length = rem as u16;
206                // little-endian: lo, hi
207                self.kallsyms_names.push((length & 0xFF) as u8);
208                self.kallsyms_names.push((length >> 8) as u8);
209            }
210
211            // Emit remaining bytes raw
212            self.kallsyms_names
213                .extend_from_slice(&sym_bytes[consumed..]);
214        }
215
216        // 4) Build name-order -> address-order sequence mapping.
217        self.kallsyms_seqs_of_names.reserve(n);
218        for &orig_idx in &idx_by_name {
219            let addr_idx = orig_to_addr_idx[orig_idx] as u32;
220            self.kallsyms_seqs_of_names.push(addr_idx);
221        }
222
223        self.kallsyms_num_syms = n;
224    }

226    /// Serialize blob into bytes
227    /// Convert the blob into binary data
228    pub fn to_blob(&self) -> Vec<u8> {
229        let mut blob = Vec::new();
230
231        #[inline]
232        fn pad(vec: &mut Vec<u8>, align: usize) {
233            let rem = vec.len() % align;
234            if rem != 0 {
235                vec.resize(vec.len() + (align - rem), 0);
236            }
237        }
238
239        blob.extend_from_slice(&(self.kallsyms_num_syms as u64).to_le_bytes());
240
241        // addresses [u64]
242        pad(&mut blob, 8);
243        for &addr in &self.kallsyms_addresses {
244            blob.extend_from_slice(&addr.to_le_bytes());
245        }
246        // offsets [u32]
247        pad(&mut blob, 4);
248        for &off in &self.kallsyms_offsets {
249            blob.extend_from_slice(&off.to_le_bytes());
250        }
251        // seqs [u32]
252        pad(&mut blob, 4);
253        for &seq in &self.kallsyms_seqs_of_names {
254            blob.extend_from_slice(&seq.to_le_bytes());
255        }
256
257        // names bytes (len u64 + bytes)
258        pad(&mut blob, 8);
259        blob.extend_from_slice(&(self.kallsyms_names.len() as u64).to_le_bytes());
260        blob.extend_from_slice(&self.kallsyms_names);
261
262        // token table bytes (len u64 + bytes)
263        pad(&mut blob, 8);
264        blob.extend_from_slice(&(self.token_table.len() as u64).to_le_bytes());
265        blob.extend_from_slice(&self.token_table);
266
267        // token index [u32] (len u64 + array)
268        pad(&mut blob, 8);
269        blob.extend_from_slice(&(self.token_index.len() as u64).to_le_bytes());
270        pad(&mut blob, 4);
271        for &idx in &self.token_index {
272            blob.extend_from_slice(&idx.to_le_bytes());
273        }
274
275        blob
276    }
277}

279fn read_symbol(line: &str) -> Option<(String, u64, char)> {
280    if line.len() > 4096 {
281        panic!("The kernel symbol is too long: {}", line);
282    }
283    let mut parts = line.split_whitespace();
284    let vaddr = u64::from_str_radix(parts.next()?, 16).ok()?;
285    let symbol_type = parts.next()?.chars().next()?;
286    let symbol = parts.collect::<Vec<_>>().join(" ");
287    Some((symbol, vaddr, symbol_type))
288}

290fn read_map() -> Vec<(String, u64, char)> {
291    let mut symbol_table = Vec::new();
292    let mut line = String::new();
293    loop {
294        let size = std::io::stdin().read_line(&mut line).unwrap();
295        if size == 0 {
296            break;
297        }
298        line = line.trim().to_string();
299        if let Some(entry) = read_symbol(&line) {
300            symbol_table.push(entry);
301        }
302        line.clear();
303    }
304    symbol_table
305}

307fn main() {
308    let symbol_table = read_map();
309    let mut blob = KallsymsBlob::new();
310    blob.compress_symbols(&symbol_table);
311    let binary_blob = blob.to_blob();
312    // Output to stdout
313    std::io::stdout()
314        .write_all(&binary_blob)
315        .expect("Failed to write blob");

317}

319#[allow(unused)]
320fn simple_test() {
321    use ksym::KSYM_NAME_LEN;
322
323    use crate::KallsymsBlob;
324
325    let symbols = r#"
326        0000000000001000 T do_mkdir
327        0000000000001300 T alias_do_fork
328        0000000000001300 T do_fork
329        0000000000001300 T do_fork_2
330        0000000000001200 T cpu_startup_entry
331        0000000000001400 T cpu_startup_entry
332        0000000000001500 T _ZN6axtask4task9TaskInner4name17h96bec46d2d8f1ab4E
333    "#;
334    let symbols: Vec<(String, u64, char)> = symbols
335        .lines()
336        .filter_map(|line| read_symbol(line.trim()))
337        .collect();
338
339    println!("Original symbols:");
340    for (sym, addr, ty) in &symbols {
341        println!("{:016x} {} {}", addr, ty, sym);
342    }
343
344    let mut blob = KallsymsBlob::new();
345    blob.compress_symbols(&symbols);
346
347    let binary_blob = blob.to_blob();
348    println!("Binary blob size: {} bytes", binary_blob.len());
349
350    let mapped = ksym::KallsymsMapped::from_blob(&binary_blob, 0x1000, 0x1600).expect("parse blob");
351
352    assert_eq!(mapped.lookup_address(100, &mut [0; KSYM_NAME_LEN]), None);
353    assert_eq!(
354        mapped.lookup_address(0x1200, &mut [0; KSYM_NAME_LEN]),
355        Some(("cpu_startup_entry", 0x100, 0, 'T'))
356    );
357    // Test aliased symbols
358    assert_eq!(
359        mapped.lookup_address(0x1300, &mut [0; KSYM_NAME_LEN]),
360        Some(("alias_do_fork", 0x100, 0, 'T'))
361    );
362    assert_eq!(
363        mapped.lookup_address(0x1250, &mut [0; KSYM_NAME_LEN]),
364        Some(("cpu_startup_entry", 0x100, 0x50, 'T'))
365    );
366    assert_eq!(
367        mapped.lookup_address(0x1450, &mut [0; KSYM_NAME_LEN]),
368        Some(("cpu_startup_entry", 0x100, 0x50, 'T'))
369    );
370    assert_eq!(
371        mapped.lookup_name("cpu_startup_entry"),
372        Some(0x1200),
373        "The address of cpu_startup_entry should be 0x1200 instead of 0x1400"
374    );
375    assert_eq!(mapped.lookup_name("do_fork_2"), Some(0x1300));
376
377    assert_eq!(
378        mapped.lookup_name("_ZN6axtask4task9TaskInner4name17h96bec46d2d8f1ab4E"),
379        Some(0x1500)
380    );
381    println!("All tests passed.");
382    let dumped = mapped.dump_all_symbols();
383    println!("Dumped all symbols:\n{}", dumped);
384}
