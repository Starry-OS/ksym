1//! A library to parse and query the Linux kernel symbol table.
2//!! It can initialize the symbol table from the embedded symbols
3//! (when compiled with `assembly` feature) or from a string in the format of
4//! `/proc/kallsyms`.
5//!
6
7#![no_std]
8#![deny(missing_docs)]
9#![feature(linkage)]

13use alloc::{
14    string::{String, ToString},
15    vec::Vec,
16};

18#[cfg(feature = "assembly")]
19pub use assembly::init_kernel_symbols;
20use lazyinit::LazyInit;
21extern crate alloc;

23/// The global kernel symbol table.
24pub static KSYM: LazyInit<Vec<(String, usize)>> = LazyInit::new();
25
26/// Initialize the kernel symbol table from a string in the format of
27/// /proc/kallsyms
28#[cfg(not(feature = "assembly"))]
29pub fn init_kernel_symbols(ksym: &str) {
30    let mut symbol_table = Vec::new();
31    for line in ksym.lines() {
32        let mut parts = line.split_whitespace();
33        let vaddr = usize::from_str_radix(parts.next().unwrap_or("0"), 16).unwrap_or(0);
34        let symbol_type = parts.next().and_then(|s| s.chars().next()).unwrap_or(' ');
35        let symbol = parts.collect::<Vec<_>>().join(" ");
36        if symbol_type == 'T' || symbol_type == 't' {
37            symbol_table.push((symbol, vaddr));
38        }
39    }
40
41    KSYM.init_once(symbol_table);
42}
43
44/// Return the function information according to the pc address.
45/// If not found, return None.
46pub fn lookup_kallsyms(addr: usize) -> Option<(String, usize)> {
47    let mut index = usize::MAX;
48    let ksym = KSYM.get().unwrap();
49    let sym_num = ksym.len();
50    for i in 0..sym_num - 1 {
51        if addr > ksym[i].1 && addr <= ksym[i + 1].1 {
52            index = i;
53            break;
54        }
55    }
56    if index < sym_num {
57        let sym_name = ksym[index].0.as_str();
58        return Some((sym_name.to_string(), ksym[index].1));
59    }
60    None
61}
62
63/// Get the address of the symbol.
64/// If not found, return None.
65pub fn addr_from_symbol(symbol: &str) -> Option<usize> {
66    let ksym = KSYM.get().unwrap();
67    for (name, addr) in ksym {
68        if name == symbol {
69            return Some(*addr);
70        }
71    }
72    None
73}
