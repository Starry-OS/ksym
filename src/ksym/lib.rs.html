1//! A library to parse and query the Linux kernel symbol table.
2//!! It can initialize the symbol table from the embedded symbols
3//! (when compiled with `assembly` feature) or from a string in the format of
4//! `/proc/kallsyms`.
5//!
6
7#![no_std]
8#![deny(missing_docs)]
9
10use alloc::string::String;
11extern crate alloc;

13/// Token delimiter marker
14pub const TOKEN_MARKER: u8 = 0xFF;
15/// Length bytes for compressed symbol
16pub(crate) const LENGTH_BYTES: usize = 2;
17
18/// Maximum length for a symbol name
19pub const KSYM_NAME_LEN: usize = 1024;
20/// Length bytes for symbol type
21const TY_LEN: usize = 1;
22
23const PREFIX_LEN: usize = LENGTH_BYTES + TY_LEN;
24/// Mapped kallsyms structure from binary blob
25pub struct KallsymsMapped<'a> {
26    token_table: &'a [u8],
27    token_index: &'a [u32],
28    kallsyms_names: &'a [u8],
29    kallsyms_offsets: &'a [u32],
30    kallsyms_seqs_of_names: &'a [u32],
31    kallsyms_addresses: &'a [u64],
32    kallsyms_num_syms: usize,
33    stext: u64,
34    etext: u64,
35}
36
37impl<'a> KallsymsMapped<'a> {
38    /// Convert binary data into the blob structure and return KallsymsMapped.
39    /// # Safety
40    /// The input blob must be well-formed and valid.
41    /// Undefined behavior may occur if the blob is malformed.
42    /// # WARNING
43    /// The blob is expected to be page-aligned in memory.
44    pub fn from_blob(blob: &'a [u8], stext: u64, etext: u64) -> Result<Self, &'static str> {
45        // page-aligned by loader (4K)
46        let base = blob.as_ptr() as usize;
47        let mut off = 0usize;
48
49        let align_off = |align: usize, off: usize| {
50            let addr = base + off;
51            let addr = (addr + align - 1) & !(align - 1);
52            addr - base
53        };
54
55        // read num_syms (u64)
56        if off + 8 > blob.len() {
57            return Err("The number of symbols is missing");
58        }
59        let num_syms = u64::from_le_bytes(blob[off..off + 8].try_into().unwrap()) as usize;
60        off += 8;
61
62        // addresses [u64], align 8
63        off = align_off(8, off);
64        let need = num_syms * core::mem::size_of::<u64>();
65        if off + need > blob.len() {
66            return Err("The addresses array is missing");
67        }
68        let addresses =
69            unsafe { core::slice::from_raw_parts(blob[off..].as_ptr() as *const u64, num_syms) };
70        off += need;
71
72        // offsets [u32], align 4
73        off = align_off(4, off);
74        let need = num_syms * core::mem::size_of::<u32>();
75        if off + need > blob.len() {
76            return Err("The offsets array is missing");
77        }
78        let offsets =
79            unsafe { core::slice::from_raw_parts(blob[off..].as_ptr() as *const u32, num_syms) };
80        off += need;
81
82        // seqs [u32], align 4
83        off = align_off(4, off);
84        let need = num_syms * core::mem::size_of::<u32>();
85        if off + need > blob.len() {
86            return Err("The seqs array is missing");
87        }
88        let seqs =
89            unsafe { core::slice::from_raw_parts(blob[off..].as_ptr() as *const u32, num_syms) };
90        off += need;
91
92        // names (len u64 + bytes), align 8
93        off = align_off(8, off);
94        if off + 8 > blob.len() {
95            return Err("The names length is missing");
96        }
97        let names_len = u64::from_le_bytes(blob[off..off + 8].try_into().unwrap()) as usize;
98        off += 8;
99        if off + names_len > blob.len() {
100            return Err("The names array is missing");
101        }
102        let names = &blob[off..off + names_len];
103        off += names_len;
104
105        // token table (len u64 + bytes), align 8
106        off = align_off(8, off);
107        if off + 8 > blob.len() {
108            return Err("The token table length is missing");
109        }
110        let token_table_len = u64::from_le_bytes(blob[off..off + 8].try_into().unwrap()) as usize;
111        off += 8;
112        if off + token_table_len > blob.len() {
113            return Err("The token table array is missing");
114        }
115        let token_table = &blob[off..off + token_table_len];
116        off += token_table_len;
117
118        // token index [u32] (len u64 + array), align 8 then 4
119        off = align_off(8, off);
120        if off + 8 > blob.len() {
121            return Err("The token index length is missing");
122        }
123        let token_index_len = u64::from_le_bytes(blob[off..off + 8].try_into().unwrap()) as usize;
124        off += 8;
125        off = align_off(4, off);
126        let need = token_index_len * core::mem::size_of::<u32>();
127        if off + need > blob.len() {
128            return Err("The token index array is missing");
129        }
130        let token_index = unsafe {
131            core::slice::from_raw_parts(blob[off..].as_ptr() as *const u32, token_index_len)
132        };
133        // off += need;
134        Ok(Self {
135            token_table,
136            token_index,
137            kallsyms_names: names,
138            kallsyms_offsets: offsets,
139            kallsyms_seqs_of_names: seqs,
140            kallsyms_addresses: addresses,
141            kallsyms_num_syms: num_syms,
142            stext,
143            etext,
144        })
145    }

147    /// Dump all symbols in address order and return as a string.
148    /// Each line: 16-digit hex address + type + name.
149    pub fn dump_all_symbols(&self) -> String {
150        let mut out = String::new();
151        let mut name_buf = [0u8; KSYM_NAME_LEN];
152
153        for i in 0..self.kallsyms_num_syms {
154            let addr = self.kallsyms_addresses[i];
155            let start = self.kallsyms_offsets[i] as usize;
156            let (len, ty) = Self::read_compressed_len_and_type(&self.kallsyms_names[start..]);
157            let end = start + PREFIX_LEN + len as usize;
158            let name =
159                self.expand_symbol(&self.kallsyms_names[start + PREFIX_LEN..end], &mut name_buf);
160            use core::fmt::Write as _;
161            let _ = writeln!(out, "{:016x} {} {}", addr, { ty }, name);
162        }
163        out
164    }

166    /// Expand a compressed symbol data into the resulting uncompressed string,
167    /// if uncompressed string is too long (>= maxlen), it will be truncated,
168    /// given the offset to where the symbol is in the compressed stream.
169    pub fn expand_symbol<'b>(
170        &self,
171        bytes: &[u8],
172        name_buf: &'b mut [u8; KSYM_NAME_LEN],
173    ) -> &'b str {
174        let mut i = 0;
175        let mut offset = 0;
176        if bytes[i] == TOKEN_MARKER {
177            let mut token_id = None;
178            // Try to parse 0xFF <id> 0xFF (1-byte) or 0xFF <id_hi> <id_lo> 0xFF (2-byte)
179            if i + 2 < bytes.len() && bytes[i + 2] == TOKEN_MARKER {
180                let id = bytes[i + 1] as u16;
181                token_id = Some(id);
182                i += 3;
183            } else if i + 3 < bytes.len() && bytes[i + 3] == TOKEN_MARKER {
184                let id = ((bytes[i + 1] as u16) << 8) | (bytes[i + 2] as u16);
185                token_id = Some(id);
186                i += 4;
187            }
188            if let Some(id) = token_id
189                && (id as usize) < self.token_index.len()
190            {
191                let start = self.token_index[id as usize] as usize;
192                let end = if (id as usize) + 1 < self.token_index.len() {
193                    self.token_index[(id as usize) + 1] as usize
194                } else {
195                    self.token_table.len()
196                };
197                let token_str = &self.token_table[start..end];
198                let need_copy = (KSYM_NAME_LEN - offset).min(token_str.len());
199                name_buf[offset..offset + need_copy].copy_from_slice(&token_str[..need_copy]);
200                offset += need_copy;
201            }
202            // Append the rest as raw bytes
203            let need_copy = (KSYM_NAME_LEN - offset).min(bytes.len() - i);
204            name_buf[offset..offset + need_copy].copy_from_slice(&bytes[i..i + need_copy]);
205            offset += need_copy;
206        } else {
207            // Not a token; treat as raw bytes
208            let need_copy = (KSYM_NAME_LEN - offset).min(bytes.len() - i);
209            name_buf[offset..offset + need_copy].copy_from_slice(&bytes[i..i + need_copy]);
210            offset += need_copy;
211        }
212
213        (core::str::from_utf8(&name_buf[..offset]).unwrap_or_default()) as _
214    }

216    fn read_compressed_len_and_type(bytes: &[u8]) -> (u16, char) {
217        if bytes.len() < PREFIX_LEN {
218            return (0, ' ');
219        }
220        // little-endian: lo first, then hi
221        // we skip the type byte
222        let len_lo = bytes[TY_LEN] as u16;
223        let len_hi = bytes[1 + TY_LEN] as u16;
224        ((len_hi << 8) | len_lo, bytes[0] as char)
225    }

227    /// Address → Symbol lookup. Returns (symbol name, symbol size, offset within symbol, type) if found.
228    ///
229    /// See <https://elixir.bootlin.com/linux/v6.6/source/kernel/kallsyms.c#L446>
230    pub fn lookup_address<'b>(
231        &self,
232        addr: u64,
233        name_buf: &'b mut [u8; KSYM_NAME_LEN],
234    ) -> Option<(&'b str, u64, u64, char)> {
235        // Quick check: address within text section and symbols exist
236        if addr < self.stext || addr >= self.etext || self.kallsyms_num_syms == 0 {
237            return None;
238        }
239        let mut low = 0usize;
240        let mut high = self.kallsyms_num_syms;
241
242        while high - low > 1 {
243            let mid = low + (high - low) / 2;
244            if self.kallsyms_addresses[mid] <= addr {
245                low = mid;
246            } else {
247                high = mid;
248            }
249        }
250
251        if low >= self.kallsyms_num_syms {
252            return None;
253        }
254
255        // Search for the first aliased symbol. Aliased
256        // symbols are symbols with the same address.
257        while low > 0 && self.kallsyms_addresses[low - 1] == addr {
258            low -= 1;
259        }
260        let symbol_start = self.kallsyms_addresses[low];
261
262        let mut symbol_end = None;
263        // Search for next non-aliased symbol.
264        for high in (low + 1)..self.kallsyms_num_syms {
265            if self.kallsyms_addresses[high] > symbol_start {
266                symbol_end = Some(self.kallsyms_addresses[high]);
267                break;
268            }
269        }
270        // If we found no next symbol, we use the end of the section.
271        let symbol_end = symbol_end.unwrap_or(self.etext);
272
273        let start = self.kallsyms_offsets[low] as usize;
274        let (len, ty) = Self::read_compressed_len_and_type(&self.kallsyms_names[start..]);
275        let end = start + PREFIX_LEN + len as usize;
276        Some((
277            self.expand_symbol(&self.kallsyms_names[start + PREFIX_LEN..end], name_buf),
278            symbol_end - symbol_start,
279            addr - symbol_start,
280            ty,
281        ))
282    }

284    /// Lookup the address for this symbol. Returns 0 if not found.
285    ///
286    /// See <https://elixir.bootlin.com/linux/v6.6/source/kernel/kallsyms.c#L265>
287    pub fn lookup_name(&self, name: &str) -> Option<u64> {
288        let (start_idx, _end_idx) = self.lookup_names(name, false)?;
289        let seq = self.kallsyms_seqs_of_names[start_idx] as usize;
290        Some(self.kallsyms_addresses[seq])
291    }

293    /// Symbol name → Address lookup (binary search on name order). It returns the first and last
294    /// name index (need_end) if there are multiple symbols with the same name. If need_end is false,
295    /// only the first index is valid.
296    ///
297    /// See <https://elixir.bootlin.com/linux/v6.6/source/kernel/kallsyms.c#L208>
298    pub fn lookup_names(&self, name: &str, need_end: bool) -> Option<(usize, usize)> {
299        let mut name_buf = [0u8; KSYM_NAME_LEN];
300        if self.kallsyms_num_syms == 0 {
301            return None;
302        }
303        let mut low = 0usize;
304        let mut high = self.kallsyms_num_syms - 1;
305        let mut mid = 0usize;
306        while low <= high {
307            mid = low + (high - low) / 2;
308            // address-order index
309            let seq = self.kallsyms_seqs_of_names[mid] as usize;
310            let start = self.kallsyms_offsets[seq] as usize;
311            let (len, _ty) = Self::read_compressed_len_and_type(&self.kallsyms_names[start..]);
312            let end = start + PREFIX_LEN + len as usize;
313            let mid_name =
314                self.expand_symbol(&self.kallsyms_names[start + PREFIX_LEN..end], &mut name_buf);
315            match name.cmp(mid_name) {
316                core::cmp::Ordering::Equal => break,
317                core::cmp::Ordering::Less => high = mid - 1,
318                core::cmp::Ordering::Greater => low = mid + 1,
319            }
320        }
321        if low > high {
322            return None;
323        }
324        low = mid;
325        // Check for earlier matches in case of duplicates
326        while low > 0 {
327            let seq = self.kallsyms_seqs_of_names[low - 1] as usize;
328            let start = self.kallsyms_offsets[seq] as usize;
329            let (len, _ty) = Self::read_compressed_len_and_type(&self.kallsyms_names[start..]);
330            let end = start + PREFIX_LEN + len as usize;
331            let mid_name =
332                self.expand_symbol(&self.kallsyms_names[start + PREFIX_LEN..end], &mut name_buf);
333            if mid_name != name {
334                break;
335            }
336            low -= 1;
337        }
338
339        high = 0;
340        if need_end {
341            // Check for later matches in case of duplicates
342            high = mid;
343            while high < self.kallsyms_num_syms - 1 {
344                let seq = self.kallsyms_seqs_of_names[high + 1] as usize;
345                let start = self.kallsyms_offsets[seq] as usize;
346                let (len, _ty) = Self::read_compressed_len_and_type(&self.kallsyms_names[start..]);
347                let end = start + PREFIX_LEN + len as usize;
348                let mid_name = self
349                    .expand_symbol(&self.kallsyms_names[start + PREFIX_LEN..end], &mut name_buf);
350                if mid_name != name {
351                    break;
352                }
353                high += 1;
354            }
355        }
356
357        Some((low, high))
358    }
